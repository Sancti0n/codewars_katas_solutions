def solution(number):
    i, somme = 1, 0
    while i<number:
        if i%3 == 0 or i%5 == 0: somme += i
        i += 1
    return somme

## PHP

function solution($number){
  $i = 1;
  $s = 0;
  while ($i<$number) {
    if ($i%3 == 0 || $i%5 == 0) $s += $i;
    $i++;
  }
  return $s;
