## JavaScript
```js
function capitalizeWord(word) {
  return word[0].toUpperCase() + word.slice(1);
}
```

## TypeScript
```ts
export function capitalizeWord(word: string): string {
  return word[0].toUpperCase() + word.slice(1);
}
```
