```js
Number.prototype.add = function(nb){    
    return this + nb;
}

Number.prototype.sub = function(nb){    
    return this - nb;
}

Number.prototype.mul = function(nb){    
    return this * nb;
}

Number.prototype.div = function(nb){    
    return this / nb;
}

Number.prototype.pow = function(nb){    
    return this**nb;
}

Number.prototype.sqr = function(){    
    return this**.5;
}
```
